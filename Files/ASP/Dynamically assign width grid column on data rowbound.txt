 protected void Gv_RowDataBound(object sender, GridViewRowEventArgs e)
 {
     try
     {
         if (e.Row.RowType == DataControlRowType.Header)
         {
             e.Row.Cells[0].Style["width"] = "4%";
             e.Row.Cells[1].Style["width"] = "48%";
             e.Row.Cells[2].Style["width"] = "48%";
         }
     }
     catch (Exception objException)
     {
         objCommonFunctions.ShowAlertMessage(objException.Message);
     }
 }
