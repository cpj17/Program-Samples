CREATE TABLE Test_EmployeeContacts (
    EmpId INT,
    ContactNum VARCHAR(15),
    FOREIGN KEY (EmpId) REFERENCES Test_Employees(EmpId)
);

CREATE TRIGGER InsertEmployeeContact
ON Test_Employees
AFTER INSERT
AS
BEGIN
    INSERT INTO Test_EmployeeContacts (EmpId, ContactNum)
    SELECT EmpId, NULL
    FROM INSERTED;
END;
